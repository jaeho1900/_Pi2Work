from openpyxl import Workbook

wb = Workbook()
ws = wb.active

ws.title = "수강생_정보"

# 시트에 추가할 칼럼의 목록을 리스트형으로 'column'이라는 변수에 지정
column = ['번호', '이름', '과목']

# column리스트 목록을 시트 첫 행에 입력
ws.append(column)

# 시트에 추가할 데이터를 리스트형으로 'row'라는 변수에 지정
row = [1, '이철수', '수학']

# append로 row의 목록을 column 아래 행에 입력
ws.append(row)

# '중간평가', '기말평가' 두 개의 시트 추가
wb.create_sheet('중간평가')
wb.create_sheet('기말평가')

wb.save("수강생_리스트.xlsx")
wb.close()


참조 -----

sorted()는 순서가 있는 자료형(리스트, 튜플)을 전달받아 정렬하는 파이썬 내장함수로 기본적으로는 오름차순으로 정렬합니다. 
key값으로 itemgetter()함수를 사용하면 리스트나 튜플, 딕셔너리에서 특정 위치의 인덱스나 키의 항목을 가져옵니다
즉, row = sorted(row, key = itemgetter(0))에서 itemgetter(0)은 리스트의 첫번째 요소(인덱스 0)를 기준으로 정렬하겠다는 의미입니다.
기준을 여러개로 지정하고자 한다면 itemgetter(1, 0) 이렇게 기준 값을 여러개로 넣어주면 됩니다.
itemgetter() 함수는 operator 모듈에 포함되어 있기 때문에, from operator import itemgetter를 통해 사용할 수 있습니다.

# 각 행의 데이터 리스트를 포함하는 리스트를 만들어 row에 저장
row = [[1, '이철수', '수학'], [3, '김미소', '영어'], [2, '최학준', '컴퓨터']]

# row리스트를 첫번째 키를 기준으로 정렬
row = sorted(row, key = itemgetter(0))

# row리스트를 for문으로 반복하여 시트에 입력
for data in row:
    ws.append(data)

>>> 판다스의 .str.contains()와 .isin() 메소드는 특정값의 포함여부를 확인할 때 사용합니다.

.str.contains()

문자열 데이터 타입에서만 적용,  '열이름'이라는 열에 '특정 단어'가 포함된 모든 행을 찾으려면,
df['열이름'].str.contains('특정 단어')

 대소문자를 구분하지 않으려면
df['열이름'].str.contains('특정 단어', case=False)

.isin()
.isin() 메소드는 모든 종류의 데이터 타입에 사용될 수 있으며, 주어진 리스트에 정확하게 일치하는 값을 찾습니다.

'열이름'이라는 열에서 '값1', '값2', '값3' 중 하나와 일치하는 모든 행을 찾으려면
df['열이름'].isin(['값1', '값2', '값3'])

하나의 값만 검사할 때도 리스트 형태로 값을 제공해야 합니다.
df['열이름'].isin(['값1'])
