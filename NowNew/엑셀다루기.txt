>>> 열 단위로 데이터 입력하기
엑셀에 열 단위로 한 번에 데이터를 입력하는 방법을 알아보겠습니다. 열 단위로 데이터를 입력하는 방법은 행 단위로 입력할 수 있는 append와 같은 함수는 없습니다. 따라서 열 단위로 입력할 때는 주로 셀에 하나씩 접근해서 입력하게 하는 방법을 사용합니다. 여기서는 열 단위로 입력할 수 있는 몇 가지 방법을 소개합니다.


1) 'cell' 메서드 사용

from openpyxl import Workbook

wb = Workbook()
ws = wb.active
ws.title = "수강생_정보"

# 열로 입력할 데이터를 리스트로 만들어 'data'에 저장
data = [ '이철수', '김미소', '최학준' ]

# for문으로 'A'열의 각 셀에 순서대로 접근하여 데이터를 입력

    ws.cell(row=i+1, column=1, value=value)

wb.save("수강생_리스트.xlsx")

# C열의 셀을 순회할 수 있는 iterator를 가져오는데, 여기서 최대 행을 'data'의 원소 갯수로 지정
col_cells = ws.iter_cols(min_col=3, max_col=3, max_row=len(data))    # 열 단위로 셀을 순회, min_col(시작 열의 인덱스 번호)', 'max_col(종료 열의 인덱스 번호) , max_row 행의 범위도 지정

# 이중for문으로 C열의 각 셀을 순환하며 'data' 리스트의 원소를 차례대로 입력
for col in col_cells:
    for i, cell in enumerate(col):
        cell.value = data[i]

# 생성된 Font 객체를 셀의 font 속성에 할당하여 셀에 글꼴 적용
# name으로 글꼴의 종류를 지정하거나 underline으로 밑줄 스타일을 설정하는 등 다양한 속성을 사용해 글꼴을 설정할 수 있습니다.
cell.font = Font(color='FF0000', 
         italic=True, 
         bold=True, 
         size=20)

#  ColumnDimension객체들이 있고 각각의 열 객체는 width(너비), hidden(숨김) 등 여러 속성을 갖습니다.
# 엑셀의 기본 너비와 높이 단위가 다르다. 행 높이 단위는 1pt(0.035cm), 열 너비는 문자 1너비(표준 글꼴 서식의 한 글자)
ws.column_dimensions['A'].width = 50

# 셀 높이 설정 (1 행의 높이를 50으로 설정)
ws.row_dimensions[1].height = 50

# 셀 배경색 설정 (노란색, 색상 코드 'FFFF00')
yellow_fill = PatternFill(start_color='FFFF00', end_color='FFFF00', fill_type='solid')

           right=thin_side,

그러면 이번에는 형식이 동일한 '11월', '12월' 시트까지 모두 동일한 작업을 실행하는 방법을 알아보겠습니다.
from openpyxl import load_workbook
from openpyxl import Workbook

# 원본 엑셀 파일 불러오기
wb = load_workbook(filename='월별구매고객리스트.xlsx', data_only=True)

# 새로운 엑셀 파일 생성
new_wb = Workbook()
new_ws = new_wb.active

# 처리할 월별 시트를 목록을 리스트 형태로 저장
months = ['10월', '11월', '12월']

# 각 월별로 반복
for month in months:
    ws = wb[month]
    index_row = [cell.value for cell in list(ws.rows)[1]]  # 원본 파일의 두 번째 행(인덱스)
    new_rows = list(ws.rows)[2:] # 처음 2개 행을 제외하고 나머지 행을 리스트로 변환

    # 데이터 분류 및 저장
    for row in new_rows:
        if row[5].value == '신규':  # F열이 '신규'인 경우
            product = row[1].value  # B열의 제품 이름 가져와 product에 할당하기

            # 해당 제품 시트가 없으면 새로 생성
            if product not in new_wb.sheetnames:
                new_wb.create_sheet(title=product)
                product_ws = new_wb[product]
                product_ws.append(index_row)  # 인덱스 행을 새 시트의 첫 번째 행에 추가

            # 해당 시트를 선택
            product_ws = new_wb[product]

            # 데이터 추가
            product_ws.append([cell.value for cell in row])

# 임시로 만든 시트 삭제
del new_wb['Sheet']

# 새로운 엑셀 파일 저장
new_wb.save(filename='제품별 신규 고객 리스트.xlsx')


동일한 형식의 엑셀 파일 여러개를 하나의 파일로 합치기
각의 파일은 "YYYY-MM-DD hhmm.xlsx"의 파일명으로 저장되어 있습니다. 

from openpyxl import Workbook, load_workbook
import os
import re

# 새로운 엑셀 파일 생성
new_wb = Workbook()
new_ws = new_wb.active
new_ws.title = "수신내역"

# 헤더 행 추가
new_ws.append(['수신시간', '발신번호', '페이지수', '용량'])

# 파일 명에 맞는 정규 표현식 패턴
pattern = re.compile(r'\d{4}-\d{2}-\d{2} \d{4}\.xlsx')

# 현재 디렉토리의 모든 파일을 순회
for filename in os.listdir('.'):
    if pattern.match(filename):
        # 엑셀 파일 불러오기
        wb = load_workbook(filename)
        ws = wb.active

        # 첫 번째 행 (헤더)은 제외하고 데이터를 새로운 엑셀 파일에 추가
        for row in ws.iter_rows(min_row=2, values_only=True):
            new_ws.append(row)
