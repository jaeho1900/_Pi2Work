from openpyxl import Workbook

# Workbook 객체 생성
wb= Workbook()

# 현재 활성화된 워크시트 선택 후 ws 변수에 할당
ws = wb.active

ws['A1'] = '이철수'

# 워크북을 '수강생_리스트.xlsx' 엑셀 파일을 저장합니다

# 워크북 닫기
wb.close()

from openpyxl import Workbook

wb = Workbook()
ws = wb.active

ws.title = "수강생_정보"

# 시트에 추가할 칼럼의 목록을 리스트형으로 'column'이라는 변수에 지정
column = ['번호', '이름', '과목']

# column리스트 목록을 시트 첫 행에 입력
ws.append(column)

# 시트에 추가할 데이터를 리스트형으로 'row'라는 변수에 지정
row = [1, '이철수', '수학']

# append로 row의 목록을 column 아래 행에 입력
ws.append(row)

# '중간평가', '기말평가' 두 개의 시트 추가
wb.create_sheet('중간평가')
wb.create_sheet('기말평가')

wb.save("수강생_리스트.xlsx")
wb.close()

>>> 엑셀 열 이름 - 열 번호 변환

열 번호로부터 엑셀 열 이름 얻기: openpyxl.utils.cell.get_column_letter(열 번호)
엑셀 열 이름으로부터 열 번호 얻기: openpyxl.utils.cell.column_index_from_string(열 이름)

from openpyxl.utils.cell import get_column_letter, column_index_from_string

# 열 번호 16에 해당하는 엑셀 열 이름 얻기
index = 16
print('열 번호 16에 해당하는 엑셀 열 이름', get_column_letter(index))   # p

# 엑셀 열 이름 'AB'에 해당하는 열 번호 얻기
print('열 이름 "AB"에 해당하는 열 번호', column_index_from_string('AB'))  # 28


>>> 열 단위로 데이터 입력하기
엑셀에 열 단위로 한 번에 데이터를 입력하는 방법을 알아보겠습니다. 열 단위로 데이터를 입력하는 방법은 행 단위로 입력할 수 있는 append와 같은 함수는 없습니다. 따라서 열 단위로 입력할 때는 주로 셀에 하나씩 접근해서 입력하게 하는 방법을 사용합니다. 여기서는 열 단위로 입력할 수 있는 몇 가지 방법을 소개합니다.


1) 'cell' 메서드 사용

from openpyxl import Workbook

wb = Workbook()
ws = wb.active
ws.title = "수강생_정보"

# 열로 입력할 데이터를 리스트로 만들어 'data'에 저장
data = [ '이철수', '김미소', '최학준' ]

# for문으로 'A'열의 각 셀에 순서대로 접근하여 데이터를 입력
for i, value in enumerate(data):
    ws.cell(row=i+1, column=1, value=value)

wb.save("수강생_리스트.xlsx")

2) 셀 주소 지정 방식 활용

from openpyxl import Workbook
from openpyxl.utils import get_column_letter

wb = Workbook()
ws = wb.active
ws.title = "수강생_정보"

# 열로 입력할 데이터를 리스트로 만들어 'data'에 저장
data = [ '이철수', '김미소', '최학준' ]

# 입력할 열의 인덱스 번호를 열이름으로 변환하여 'column_name'에 저장
column_name = get_column_letter(2)

for i, value in enumerate(data):
    # f-string을 사용하여 셀 주소를 지정하고 값을 할당
    ws[f"{column_name}{i+1}"] = value

wb.save("수강생_리스트.xlsx")

3) iter_cols() 함수 사용

from openpyxl import Workbook

wb = Workbook()
ws = wb.active
ws.title = "수강생_정보"

# 열로 입력할 데이터를 리스트로 만들어 'data'에 저장
data = [ '이철수', '김미소', '최학준' ]

# C열의 셀을 순회할 수 있는 iterator를 가져오는데, 여기서 최대 행을 'data'의 원소 갯수로 지정
col_cells = ws.iter_cols(min_col=3, max_col=3, max_row=len(data))    # 열 단위로 셀을 순회, min_col(시작 열의 인덱스 번호)', 'max_col(종료 열의 인덱스 번호) , max_row 행의 범위도 지정

# 이중for문으로 C열의 각 셀을 순환하며 'data' 리스트의 원소를 차례대로 입력
for col in col_cells:
    for i, cell in enumerate(col):
        cell.value = data[i]

wb.save("수강생_리스트.xlsx")

4) 행과 열의 인덱스 번호로 데이터 입력

from openpyxl import Workbook

wb = Workbook()
ws = wb.active
ws.title = "수강생_정보"

# A열과 B열에 입력할 데이터를 이중리스트로 만들어 'data'에 저장
data = [
    [ '이철수', '김미소', '최학준' ], [ '수학', '영어', '컴퓨터' ]
]

# 이중for문으로 열의 인덱스 번호와 행의 인덱스 번호로 셀에 데이터를 입력
for col_idx, col_data in enumerate(data, start=1): 
    for row_idx, row_data in enumerate(col_data):
        ws.cell(row=row_idx+1, column=col_idx, value=row_data)

wb.save("수강생_리스트.xlsx")

팁.데이터를 입력할 열과 열 사이에 공백을 넣고자 한다면, 이중리스트에서 각 리스트 사이에 원하는 공백 열의 갯수만큼 빈 리스트 '[]' 넣어주면 그만큼 열을 건너뛰고 그 다음 열에 데이터를 입력합니다.

>>> 특정 행을 기준으로 데이터 가져오기
# rows()로 데이터 가져오기
# 첫번째(빈행)와 두번째 행(제목줄행)을 제외하고 나머지 행을 리스트로 변환
new_rows = list(ws.rows)[2:]
# 1)나머지 행을 순회하며 읽기
for row in new_rows:
    row_values = [cell.value for cell in row]
    print(row_values)

# 2)iter_rows로 데이터 가져오기
# 3번째 행부터 마지막 행까지를 순회하며 값을 읽기 (min_row =3)
for row in ws.iter_rows(min_row=3, values_only=True):   # values_only는 각 셀 객체 대신 셀의 값만 반환할지 여부를 설정할 수 있는 매개변수
    print(row)
